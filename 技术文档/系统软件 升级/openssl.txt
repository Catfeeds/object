#先进行支撑包的安装：

yum install -y zlib 



#openssl升级步骤： 



#首先通过 #openssl version Ca 查看系统中存在的OpenSSL版本号 

#下载最新版本的openssl源码包 

# wget ftp://ftp.openssl.org/source/openssl-1.0.1g.tar.gz

#. 安装openssl 

 

tar Cxzvf openssl-1.0.1g.tar.gz  

cd openssl-1.0.1g  

./config --prefix=/usr/local/ssl  shared zlib 

make  

make install  

mv /usr/bin/openssl /usr/bin/openssl.0.9.8e 

mv /usr/include/openssl /usr/include/openssl.0.9.8e


ln -s /usr/local/ssl/bin/openssl /usr/bin/openssl

ln -s /usr/local/ssl/include/openssl/ /usr/include/openssl

#配置库文件搜索路径

echo “/usr/local/ssl/lib” >> /etc/ld.so.conf 

ldconfig Cv  



#查看openssl版本号，验证安装正确性 



openssl version Ca
